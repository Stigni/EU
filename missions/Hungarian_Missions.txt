vassalize_transylvania_hun = {
	
	type = country

	category = MIL
	ai_mission = yes
	
	allow = {
		normal_or_historical_nations = yes
		tag = HUN
		exists = TRA
		NOT = { has_country_modifier = military_vassalization }
		is_subject = no
		TRA = {
			is_subject = no
			is_neighbor_of = ROOT
			NOT = { num_of_cities = ROOT }
			NOT = { total_development = 50 }
		}
	}
	abort = {
		OR = {
			is_subject = yes
			NOT = { exists = TRA }
			TRA = { is_subject = yes }
			NOT = {
				has_casus_belli = {
					type = cb_vassalize_mission
					target = TRA
				}
			}
		}
	}
	success = {
		TRA = { vassal_of = HUN }
	}
	chance = {
		factor = 1000
		modifier = {
			factor = 2
			NOT = { has_opinion = { who = TRA value = 100 } }
		}
		modifier = {
			factor = 2
			NOT = { has_opinion = { who = TRA value = 200 } }
		}
	}
	immediate = {
		add_casus_belli = {
			type = cb_vassalize_mission
			months = 300
			target = TRA
		}
	}
	abort_effect = {
		remove_casus_belli = {
			type = cb_vassalize_mission
			target = TRA
		}
	}
	effect = {
		add_country_modifier = {
			name = military_vassalization
			duration = 3650
		}
		hidden_effect = {
			remove_casus_belli = {
				type = cb_vassalize_mission
				target = TRA
			}
		}
	}
}
