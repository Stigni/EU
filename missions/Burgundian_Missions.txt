conquer_hainaut = {
	
	type = country

	category = MIL
	ai_mission = yes
	
	target_provinces = {
		owned_by = HAI
	}
	allow = {
		normal_or_historical_nations = yes
		tag = BUR
		is_subject = no
		exists = HAI
		HAI = {
			is_subject = no
		}
		NOT = { alliance_with = HAI }
	}
	abort = {
		is_subject = yes
	}
	success = {
		NOT = { exists = HAI }
		all_target_province = {
			owned_by = ROOT
		}
	}
	chance = {
		factor = 3000
		modifier = {
			factor = 2
			NOT = { has_opinion = { who = HAI value = 0 } }
		}	
	}
	immediate = {
		every_target_province = {
			add_claim = ROOT
		}
	}
	abort_effect = {
		every_target_province = {
			remove_claim = ROOT
		}
	}
	effect = {
		every_target_province = {
			add_province_modifier = {
				name = "faster_integration"
				duration = 3650
			}
		}
	}
}

vassalize_lorraine = {
	
	type = country

	category = MIL
	ai_mission = yes
	
	allow = {
		normal_or_historical_nations = yes
		tag = BUR
		exists = LOR
		is_subject = no
		NOT = { has_country_modifier = military_vassalization }
		LOR = {
			is_subject = no
			is_neighbor_of = ROOT
			NOT = { alliance_with = ROOT }
			NOT = { num_of_cities = ROOT }
			NOT = { total_development = 50 }
		}
	}
	abort = {
		OR = {
			is_subject = yes
			NOT = { exists = LOR }
			LOR = { is_subject = yes }
			NOT = {
				has_casus_belli = {
					type = cb_vassalize_mission
					target = LOR
				}
			}
		}
	}
	success = {
		LOR = { vassal_of = BUR }
	}
	chance = {
		factor = 3000
		modifier = {
			factor = 2
			NOT = { has_opinion = { who = LOR value = 0 } }
		}
	}
	immediate = {
		add_casus_belli = {
			type = cb_vassalize_mission
			months = 300
			target = LOR
		}
	}
	abort_effect = {
		remove_casus_belli = {
			type = cb_vassalize_mission
			target = LOR
		}
	}
	effect = {
		add_country_modifier = {
			name = military_vassalization
			duration = 3650
		}
		hidden_effect = {
			remove_casus_belli = {
				type = cb_vassalize_mission
				target = LOR
			}
		}
	}
}
