		}

			owned_by = ROOT
		}
	}

		modifier = {
			factor = 3

		}

			factor = 3
			mil = 6
		}
	}

		every_target_province = {
			add_claim = ROOT
		}
	}
	abort_effect = {
		every_target_province = {
			remove_claim = ROOT
		}
	}
	effect = {
		add_country_modifier = {
			name = "military_victory"
			duration = 3650
		}
	}
