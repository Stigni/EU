unite_austria = {
	
	type = country

	category = MIL
	ai_mission = yes
	
	target_provinces = {

			owned_by = STY

		}

		normal_or_historical_nations = yes

	abort = {
		OR = {

				NOT = { exists = STY }
				NOT = { exists = TIR }
			}
		}
	}
	success = {
		NOT = { exists = STY }
		NOT = { exists = TIR }
		all_target_province = {
			owned_by = ROOT
		}
	}
	chance = {
		factor = 3000
		modifier = {
			factor = 2
			TIR = { has_opinion = { who = HAB value = 100 } }
		}
		modifier = {
			factor = 2
			STY = { has_opinion = { who = HAB value = 100 } }
		}
	}
	immediate = {
		every_target_province = {
			add_claim = ROOT
		}
	}
	abort_effect = {
		every_target_province = {
			remove_claim = ROOT
		}
	}
	effect = {
		set_country_flag = austria_reunited_flag

		}
