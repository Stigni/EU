defend_the_american_colonies = {
	
	type = country
	
	category = MIL

	allow = {
		normal_or_historical_nations = yes
		tag = USA
		is_subject = no
		the_thirteen_colonies = {
			owned_by = USA
		}
		the_thirteen_colonies = {
			owned_by = GBR
		}
	}
	abort = {
		is_subject = yes
	}
	success = {
		NOT = { war_with = GBR }
		the_thirteen_colonies = {
			type = all
			owned_by = USA
		}
	}
	chance = {
		factor = 3000
		modifier = {
			factor = 2
			mil = 4
		}
		modifier = {
			factor = 2
			ENG = { NOT = { mil = 2 } }
		}
	}
	effect = {
		add_army_tradition = 50
		add_stability = 2
	}
}
