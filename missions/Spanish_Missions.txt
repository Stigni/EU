finish_reconquista = {

	type = country

	category = MIL
	ai_mission = yes
	
	target_provinces = {
		region = iberia_region
		owner = { religion_group = muslim }
	}
	allow = {
		normal_or_historical_nations = yes
		NOT = { has_country_modifier = completing_the_reconquista }
		NOT = { has_country_flag = granada_gone }
		NOT = { is_year = 1600 }
		OR = {
			tag = CAS
			tag = SPA
		}
		is_subject = no
		is_at_war = no
		religion = catholic
		iberia_region = {
			owned_by = ROOT
		}
		iberia_region = {
			owner = { religion_group = muslim }
		}
	}
	abort = {
		OR = {
			NOT = { religion = catholic }
			NOT = { iberia_region = { owned_by = ROOT } }
		}
	}
	success = {
		iberia_region = {
			type = all
			owner = { NOT = { religion_group = muslim } }
		}
		all_target_province = {
			owned_by = ROOT
		}
	}
	immediate = {
		every_target_province = {
			add_claim = ROOT
		}
	}
	abort_effect = {
		every_target_province = {
			if = {
				limit = {
					NOT = {
						is_permanent_claim = ROOT
					}
				}
				remove_claim = ROOT
			}
		}
	}
	chance = {
		factor = 3000
		modifier = {
			factor = 10
			exists = GRA
			iberia_region = {
				owned_by = GRA
			}
		}
		modifier = {
			factor = 1000
			is_tutorial_active = yes
		}
	}
	effect = {
		set_country_flag = granada_gone
		add_country_modifier = {
			name = "completing_the_reconquista"
			duration = 3650
		}
	}
}

spain_must_be_christian = {

		}

		is_at_war = no
		NOT = {	exists = GRA }
		iberia_region = {
			owned_by =  ROOT
			NOT = { religion_group = christian }
		}
		NOT = { has_country_modifier = achieved_religious_unity_mission }
	}
	abort = {
		NOT = {	religion = catholic  }
	}	
	success = {
		NOT = {
			any_owned_province = {
				region = iberia_region
				NOT = { religion_group = christian }
			}
		}
	}
	chance = {
		factor = 3000
		modifier = {
			factor = 2
			adm = 3
		}
	}
	effect = {
		add_country_modifier = {
			name = "achieved_religious_unity_mission"
			duration = 3650
		}
		if = {
			limit = {
				any_owned_province = {
					region = iberia_region
					any_core_country = {
						NOT = { religion_group = christian }
					}
				}
			}
			every_owned_province = {
				limit = {
					region = iberia_region
					any_core_country = {
						NOT = { religion_group = christian }
					}
				}
				every_core_country = {
					limit = {
						NOT = { religion_group = christian }
					}
					remove_core = PREV
				}
			}
		}
	}
}
