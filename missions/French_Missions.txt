subjugate_brittany = {
	
	type = country

	category = MIL
	ai_mission = yes
	
	allow = {
		normal_or_historical_nations = yes
		NOT = { has_country_modifier = military_vassalization }
		has_country_flag = had_end_of_HYW_france
		tag = FRA
		is_subject = no
		BRI = {
			is_subject = no
			is_neighbor_of = ROOT
			NOT = { alliance_with = ROOT }
			NOT = { num_of_cities = ROOT }
			NOT = { total_development = 60 }
			NOT = { government = imperial_city }
		}
	}
	abort = {
		OR = {
			is_subject = yes
			NOT = {
				has_casus_belli = {
					type = cb_vassalize_mission
					target = BRI
				}
			}
		}
	}
	success = {
		BRI = { vassal_of = ROOT }
	}
	chance = {
		factor = 2000
		modifier = {
			factor = 2
			NOT = { has_opinion = { who = BRI value = 0 } }
		}
	}
	immediate = {
		add_casus_belli = {
			type = cb_vassalize_mission
			months = 300
			target = BRI
		}
	}
	abort_effect = {
		remove_casus_belli = {
			type = cb_vassalize_mission
			target = BRI
		}
	}
	effect = {
		add_country_modifier = {
			name = military_vassalization
			duration = 3650
		}
		hidden_effect = {
			remove_casus_belli = {
				type = cb_vassalize_mission
				target = BRI
			}
		}
	}
}
