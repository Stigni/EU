get_minor_cash_reserve = {

	type = country

	category = ADM

	allow = {
		NOT = { years_of_income = 0.5 }
		NOT = { num_of_loans = 1 }
		NOT = { has_country_flag = got_minor_cash_reserve }
	}
	abort = {
		num_of_loans = 1
	}
	success = {
		years_of_income = 1.0
	}
	chance = {
		factor = 1000
	}
	effect = {
		add_prestige = 10
		set_country_flag = got_minor_cash_reserve
	}	
}


amass_wealth = {

	type = country
	
	category = ADM

	allow = {
		NOT = { treasury = 1000 }
		monthly_income = 100
		treasury = 300
		is_at_war = no
		NOT = { num_of_loans = 1 }
		NOT = { has_country_flag = amassed_wealth }
	}
	abort = {
		OR = {
			NOT = { treasury = 200 }
			num_of_loans = 1
		}
	}
	success = {
		treasury = 2000
	}
	chance = {
		factor = 1000
		modifier = {
			factor = 1.02
			inflation = 1
		}		
		modifier = {
			factor = 1.05
			full_idea_group = economic_ideas
		}		
		modifier = {
			factor = 1.02
			full_idea_group = administrative_ideas
		}		
	}
	effect = {
		add_prestige = 10
		add_inflation = -1
		set_country_flag = amassed_wealth
	}	
}
