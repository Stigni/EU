aspiration_for_liberty = {
	potential = {
		NOT = { has_country_flag = had_liberalism }
		NOT = { current_size_of_parliament = 1 }
		uses_doom = no
		technology_group = western
		num_of_cities = 5
		is_year = 1650
		is_subject = no
	}

	
	
	can_start = {
		has_any_disaster = no
		adm_tech = 22
		NOT = { stability = 0 }
	}
	
	can_stop = { 
		OR = {
			has_any_disaster = yes
			stability = 3
		}
	}
	
	
	progress = {
		modifier = {
			factor = 0.5
			government = absolute_monarchy
		}
		modifier = {
			factor = 0.5
			government = enlightened_despotism
		}
		modifier = {
			factor = 0.5
			government = bureaucratic_despotism
		}
		modifier = {
			factor = 0.5
			government = republican_dictatorship
		}
		modifier = {
			factor = 1
			is_bankrupt = yes
		} 
		modifier = {
			factor = 1
			war_exhaustion = 5
		} 		

	}
	

	
	can_end = {
		OR = {
			current_size_of_parliament = 1
			AND = {
				stability = 1
				NOT = { num_of_rebel_controlled_provinces = 1 }
			}
		}
	}
	

	modifier = {
		global_unrest = 7
		legitimacy = -3
	}	
	
	
	on_start = aspiration_for_liberty.1
	on_end = aspiration_for_liberty.4
	
	on_monthly = {
		events = {
		}
		random_events = { 
			1000 = 0
			100 = aspiration_for_liberty.2
			100 = aspiration_for_liberty.3
		
		}
	}
}
