# Election!
country_event = {
	id = 700
	title = "EVTNAME700"
	desc = "EVTDESC700"
	picture = ELECTION_REPUBLICAN_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		#is_colonial_nation = no
		NOT = { government = dutch_republic }
	}

	option = {
		name = "EVTOPTA700"	# Current ruler stays
		trigger = {
			has_regency = no
		}
		ai_chance = { 
			factor = 60
			modifier = {
				factor = 0
				NOT = { republican_tradition = 0.25 }
			}	
			modifier = {
				factor = 0.5
				NOT = { republican_tradition = 0.5 }
			}				
			modifier = {
				factor = 0.5
				NOT = { republican_tradition = 0.75 }
			}	
			modifier = {
				factor = 2.0
				republican_tradition = 0.9
			}		
		}
		custom_tooltip = remains_ruler
		change_adm = 1
		change_dip = 1
		change_mil = 1
		if = {
			limit = { 
				is_tribal = yes
				government = republic
				NOT = { republican_tradition = 0.20 }	
			}	
			change_government = tribal_despotism
		}
		if = {
			limit = {
				is_tribal = no 
				government = republic
				NOT = { republican_tradition = 0.20 }				
				has_dlc = "Res Publica"
			}
			change_government = republican_dictatorship	
		}
		if = {
			limit = {
				is_tribal = no  
				government = republic
				NOT = { republican_tradition = 0.20 }
				NOT = { has_dlc = "Res Publica" }
			}
			change_government = despotic_monarchy
		}	
		add_scaled_republican_tradition = -0.10
		random_list = {
			10 = {
				add_adm_power = 50
			}
			10 = {
				add_dip_power = 50
			}
			10 = {
				add_mil_power = 50
			}
		}
	}
	option = {
		name = "EVTOPTB700"	# An administrator type
		ai_chance = { factor = 20 }
		define_ruler = {
			mil = 1
			adm = 4
			dip = 1					
			fixed = yes
		}
		if = {
			limit = { 
				has_faction = mr_guilds 
			}
			add_faction_influence = {
				faction = mr_guilds
				influence = 20
			}				
		}
	}
	option = {
		name = "EVTOPTC700"	# A diplomat
		ai_chance = { factor = 20 }
		define_ruler = {
			dip = 4
			adm = 1
			mil = 1					
			fixed = yes
		}
		if = {
			limit = { 
				has_faction = mr_traders 
			}
			add_faction_influence = {
				faction = mr_traders
				influence = 20
			}				
		}		
	}
	option = {
		name = "EVTOPTD700"	# A military man
		ai_chance = { factor = 20 }
		define_ruler = {
			mil = 4
			adm = 1
			dip = 1					
			fixed = yes
		}
		if = {
			limit = { 
				has_faction = mr_aristocrats 
			}
			add_faction_influence = {
				faction = mr_aristocrats
				influence = 20
			}				
		}		
	}
}
