# Protestant Faith?
country_event = {
	id = westernisation.11
	title = "EVTNAME6405"
	desc = "EVTDESC6405"
	picture = RELIGION_eventPicture

	is_triggered_only = yes
	
	trigger = {
		OR = {
			religion = animism
			religion = shamanism
			religion = totemism
		}
		is_westernising = yes
		any_owned_province = {
			any_neighbor_province = {
				owner = { technology_group = western }
				religion = protestant
			}
		}
		NOT = { has_country_flag = refused_protestant }
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = "EVTOPTA6405"
		change_religion = protestant
	}
    option = {
		name = "EVTOPTB6405"
		add_westernisation = -120
		set_country_flag = refused_protestant
	}
}
