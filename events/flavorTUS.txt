#The emperor condemns the Duchy's elevation
country_event = {
	id = flavor_tus.2
	title = flavor_tus.2.t
	desc = flavor_tus.2.d
	picture = REFORM_eventPicture

	fire_only_once = yes
	
	trigger = {
		tag = TUS
		NOT = {
			is_year = 1600
		}
		is_emperor = no
		NOT = {
			reverse_has_opinion = {
				who = emperor
				value = 75 
			}
		}
		reverse_has_opinion = {
			who = PAP
			value = 50
		}
		religion = catholic
		NOT = {
			num_of_cities = 6
		}
		government = monarchy
	}

	option = {
		name = flavor_tus.2.a
		ai_chance = {
			factor = 80
		}
		emperor = {
			add_opinion = {
				who = TUS
				modifier = opinion_emperor_tus_refused
			}
		}
		PAP = {
			add_opinion = {
				who = TUS
				modifier = opinion_papal_tus_refused
			}
		}
		add_legitimacy = -10
		add_prestige = 10
		add_country_modifier = {
			name = emperor_condemns_tuscany
			duration = 7300
		}
	}

	option = {
		name = flavor_tus.2.b 	#We yield to the emperor
		add_prestige = -15
		emperor = {
			add_opinion = {
				who = TUS
				modifier = opinion_emperor_tus_yielded
			}
		}
		PAP = {
			add_opinion = {
				who = TUS
				modifier = opinion_papal_tus_yielded
			}
		}
	}
}

#Savonarola Rises
country_event = {
	id = flavor_tus.9
	title = flavor_tus.9.t
	desc = flavor_tus.9.d
	picture = RELIGIOUS_TURMOIL_eventPicture

	fire_only_once = yes

	trigger = {
		OR = {
			tag = LAN
			tag = TUS
		}
		NOT = {
			is_year = 1520
		}
		is_year = 1485
		capital = 116
		OR = {
			NOT = {
				republican_tradition = 0.85
			}
			government = monarchy
		}
	}

	mean_time_to_happen = {
		months = 300
		modifier = {
			factor = 0.8
			is_at_war = yes
			NOT = {
				war_score = 10
			}
		}
		modifier = {
			factor = 0.7
			OR = {
				AND = {
					government = republic
					NOT = {
						republican_tradition = 0.5
					}
				}
				government = monarchy
			}
		}
		modifier = {
			factor = 0.7
			any_country = {
				region = italy_region
				OR = {
					FRA = {
						war_with = PREV
					}
					HAB = {
						war_with = PREV
					}
					SPA = {
						war_with = PREV
					}
				}
			}
		}
		modifier = {
			factor = 1.3
			stability = 2
		}
		modifier = {
			factor = 0.7
			NOT = {
				stability = 0
			}
		}
	}

	#Ruling Family Flees!
	option = {
		name = flavor_tus.9.a
		ai_chance = {
			factor = 75
		}
		define_ruler = {
			name = "Girolamo"
			dynasty = "Savonarola"
			age = 40
			ADM = 4
			DIP = 2
			MIL = 2
		}
		add_republican_tradition = 0.20
		set_country_flag = tus_savonarola_seized_power
	}

	#Savonarola Must Be Arrested!
	option = {
		name = flavor_tus.9.b
		ai_chance = {
			factor = 25
		}
		add_stability = -1
		capital_scope = {
			spawn_rebels = {
				type = catholic_rebels
				size = 3
			}
		}
		add_republican_tradition = -0.2
	}
}

#The People Turn Against Savonarola
country_event = {
	id = flavor_tus.15
	title = flavor_tus.15.t
	desc = flavor_tus.15.d
	picture = LIBERUM_VETO_eventPicture

	trigger = {
		had_country_flag = {
			flag = tus_savonarola_seized_power
			days = 365
		}
	}

	mean_time_to_happen = {
		months = 48
		modifier = {
			factor = 0.8
			has_country_modifier = tus_bonfire_timer
		}
	}

	immediate = {
		clr_country_flag = tus_savonarola_seized_power
	}

	#Let him walk the Trial by Fire
	option = {
		name = flavor_tus.15.a
		random_list = {
			98 = {
				if = {
					limit = {
						NOT = {
							government = republic
						}
					}
					change_government = oligarchic_republic
				}
				define_ruler = {
					dynasty = "de' Medici"
				}
			}
			2 = {
				change_government = theocratic_government
			}
		}
	}

	#The people will never again challenge God's command!
	option = {
		name = flavor_tus.15.b
		add_stability = -1
		add_republican_tradition = -0.5
	}
}
