# The Effect of the Pragmatic Sanction

	id = pragmatic_sanction.1
	title = "pragmatic_sanction.EVTNAME1"
	desc = "pragmatic_sanction.EVTDESC1"
	picture = REFORM_eventPicture

	trigger = {
		NOT = { has_country_flag = pragmatic_sanction }
		HAB = {
			exists = yes
			has_female_heir = yes
			is_emperor = yes
			has_country_flag = pragmatic_sanction
		}
		OR = {
			tag = FRA
			tag = SWE
			tag = BAV
			tag = DAN
			tag = PRU
			tag = SAX
			OR = {
				AND = {
					tag = CAS
					NOT = { exists = SPA }
				}
				tag = SPA
			}
			OR = {
				AND = {
					tag = ENG
					NOT = { exists = GBR }
				}
				tag = GBR
			}
			OR = {
				AND = {
					tag = MOS
					NOT = { exists = RUS }
				}
				tag = RUS
			}
			OR = {
				AND = {
					tag = HOL
					NOT = { exists = NED }
				}
				tag = NED
			}
		}
	}

	mean_time_to_happen = {
		months = 12
	}

	immediate = {
		set_country_flag = pragmatic_sanction
	}

	option = {		# Pro-Habsburg
		name = "pragmatic_sanction.EVTOPTA1"
		ai_chance = { factor = 25 }
		add_stability = 1
		HAB = {
			add_opinion = {
				who = ROOT
				modifier = pro_pragmatic_sanction
			}
		}
	}
	option = {		# Neutral
		name = "pragmatic_sanction.EVTOPTB1"
		ai_chance = { factor = 25 }
		add_stability = -1
	}
	option = {		# Anti-Habsburg
		name = "pragmatic_sanction.EVTOPTC1"
		ai_chance = { factor = 50 }
		add_stability = 1
		HAB = {
			add_opinion = {
				who = ROOT
				modifier = anti_pragmatic_sanction
			}
		}
	}
}
