# The Marathi Revolt
country_event = {
	id = flavor_dlh.1
	title = "flavor_dlh.EVTNAME1"
	desc = "flavor_dlh.EVTDESC1"
	picture = ANGRY_MOB_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		religion_group = muslim
		is_year = 1650
		NOT = { tag = MUG }
		NOT = { exists = MAR } # Marathas
		any_owned_province = {
			OR = {
					area = maharashtra_area
					area = desh_area
					culture = marathi
				}
		}
	}
	
	mean_time_to_happen = {
		months = 200
	}
	
	option = {
		name = "flavor_dlh.EVTOPTA1"
		every_owned_province = {
			limit = {
				OR = {
					area = maharashtra_area
					area = desh_area
					culture = marathi
				}
			}
			add_core = MAR
			add_unrest = 10
		}
	}
	option = {
		name = "flavor_dlh.EVTOPTB1"
		every_owned_province = {
			limit = {
				OR = {
					area = maharashtra_area
					area = desh_area
					culture = marathi
				}
				is_capital = no
			}
			add_local_autonomy = 100
		}
	}
}

# The Lodi Dynasty
country_event = {
	id = flavor_dlh.3
	title = "flavor_dlh.EVTNAME3"
	desc = "flavor_dlh.EVTDESC3"
	picture = COURT_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = DLH
		dynasty = "Lodi" 
		OR = { 
			NOT = { dip = 2 } 
			NOT = { mil = 2 } 
			NOT = { adm = 2 }
			NOT = { legitimacy = 70 }
			NOT = { stability = 1 }
		}
	}
	
	mean_time_to_happen = {
		months = 200
	}
	
	option = {
		name = "flavor_dlh.EVTOPTA3"
		add_stability = -1
		add_inflation = 2
		set_country_flag = dlh_lodi_flag
	}
}
