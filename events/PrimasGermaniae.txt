# Title of Primas Germaniae Questioned
country_event = {
	id = 760
	title = "EVTNAME760"
	desc = "EVTDESC760"
	picture = POPE_PREACHING_eventPicture
	
	trigger = {
		papacy_active = yes
		has_country_modifier = primas_germaniae
		NOT = { prestige = -10 }
		any_country = {
			is_emperor = no
			is_elector = yes
			government = theocratic_government
			religion = catholic
			NOT = { has_opinion = { who = ROOT value = 0 } }
		}
	}

	mean_time_to_happen = {
		months = 6
		modifier = {
			factor = 1.5
			papal_influence = 20 
		}
		modifier = {
			factor = 2
			papal_influence = 35
		}
		modifier = {
			factor = 0.75
			NOT = { has_opinion = { who = PAP value = 100 } }
		}
		modifier = {
			factor = 0.5
			NOT = { has_opinion = { who = PAP value = 0 } }
		}
		modifier = {
			factor = 10
			has_global_flag = primas_germaniae_no_candidate
		}
	}

	option = {
		name = "EVTOPTA760"
		hidden_effect = {
			random_elector = {
				limit = {
					is_emperor = no
					government = theocratic_government
					religion = catholic
					NOT = { has_opinion = { who = ROOT value = 100 } }
					prestige = 10
					NOT = { has_country_flag = primas_germaniae_declined }
				}
				set_global_flag = primas_germaniae_candidate_found
				country_event = { id = 761 days = 3 }
			}
			if = {
				limit = {
					NOT = { has_global_flag = primas_germaniae_candidate_found }
				}
				set_global_flag = primas_germaniae_no_candidate
			}
		}		
	}
}
