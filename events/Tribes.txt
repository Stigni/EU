# Get a claim on a neighbouring richer province
province_event = {
	id = 9200
	title = "EVTNAME9200"
	desc = "EVTDESC9200"
	picture = DEBATE_REPUBLICAN_eventPicture

	trigger = {
		owner = {
			government = steppe_horde
			is_at_war = no
			num_of_cities = 10
		}
		is_core = ROOT
		any_neighbor_province = {
			NOT = { owned_by = ROOT }
			NOT = { is_core = ROOT }
			NOT = { is_claim = ROOT }
			development = 10
			owner = { is_tribal = no }
		}
	}
	
	mean_time_to_happen = {
	
		months = 1800

		modifier = {
			factor = 0.8
			owner = { mil = 5 }
		}
	}

	option = {
		name = "EVTOPTA9200"
		random_neighbor_province = {
			limit = {
				NOT = { owned_by = ROOT }
				NOT = { is_core = ROOT }
				development = 10
				owner = { is_tribal = no }
			}
			add_claim = ROOT
		}
		
	}
}

# Get conquest CB on a weaker neighbour
country_event = {
	id = 9201
	title = "EVTNAME9201"
	desc = "EVTDESC9201"
	picture = DIPLOMACY_eventPicture

	trigger = {
		always = no # unnecessary, already has CB
		government = steppe_horde
		is_at_war = no
		num_of_cities = 10
		any_neighbor_country = {
			NOT = { num_of_cities = 4 }	
			NOT = { has_country_flag = is_tribal_target }		
		}
	}
	
	mean_time_to_happen = {
		months = 500
		modifier = {
			factor = 0.8
			mil = 5
		}
	}

	option = {
		name = "EVTOPTA9201"
		random_country = {
			limit = {
				is_neighbor_of = ROOT
				NOT = { num_of_cities = 4 }
				NOT = { has_country_flag = is_tribal_target }				
			}
			set_country_flag = is_tribal_target
			capital_scope = { add_claim = ROOT }
		}
		
	}
}

# Get a pretender rising if low war-exhaustion and not in a war
country_event = {
	id = 9202
	title = "EVTNAME9202"
	desc = "EVTDESC9202"
	picture = CIVIL_WAR_eventPicture

	trigger = {
		government = steppe_horde
		is_at_war = no
		num_of_cities = 10
		NOT = { war_exhaustion = 1 }
		has_regency = no
		NOT = { num_of_rebel_controlled_provinces = 1 }
		NOT = { had_recent_war = 10 }
		has_truce = no
		NOT = { has_spawned_rebels = pretender_rebels }
	}
	
	mean_time_to_happen = {
		months = 12
		modifier = {
			factor = 0.5
			mil = 4
		}
		modifier = {
			factor = 10
			ai = yes
		}
	}
	
	option = {
		name = "EVTOPTA9202"
		random_owned_province = {
			pretender_rebels = 1
		}
	}
}

			}
